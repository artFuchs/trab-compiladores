PROG = DECLARACOES

DECLARAÇÕES =
    DECL_FUNÇÃO; DECLARAÇÕES
    | DECL_VARIÁVEL; DECLARAÇÕES 
    | nada

DECL_VARIÁVEL = 
    TIPO ID = VALOR
    | TIPO ID [ VALORINT ] INIVETOR
    
INIVETOR =
    : VALOR INIVETOR'
    | nada
    
INIVETOR' = 
    , VALOR INIVETOR' 
    | nada


DECL_FUNÇÃO = TIPO ID (PARAMS) BLOCO
PARAMS = 
    TIPO NOME, PARAMS
    | nada

BLOCO = { BL_COMANDOS }

BL_COMANDOS = 
    COMANDO; BL_COMANDOS
    | COMANDO
    | nada
    
COMANDO = 
    ID = EXPRESSAO
    | ID [ EXPRESSAO ] = EXPRESSAO
    | CONTROLE_FLUXO
    | read ID
    | print ELEMENTOS
    | return EXPRESSAO
    | BLOCO
    
ELEMENTOS = 
    STRING ELEMENTOS'
    | EXPRESSAO ELEMENTOS'
    
ELEMENTOS =
    , STRING ELEMENTOS'
    | , EXPRESSAO ELEMENTOS'
    | nada
    
CONTROLE_FLUXO =
    if (EXPRESSAO) then COMANDO else COMANDO
    | if (EXPRESSAO) then COMANDO
    | loop (EXPRESSAO) COMANDO
    | leap
    
EXPRESSAO =
    ID
    | ID [ EXPRESSAO ]
    | LITERAL
    | ID ( ARGUMENTOS )
    | EXPRESSAO + EXPRESSAO
    | EXPRESSAO - EXPRESSAO
    | EXPRESSAO * EXPRESSAO
    | EXPRESSAO / EXPRESSAO
    | EXPRESSAO < EXPRESSAO
    | EXPRESSAO > EXPRESSAO
    | EXPRESSAO and EXPRESSAO
    | EXPRESSAO or EXPRESSAO
    | EXPRESSAO not EXPRESSAO
    | EXPRESSAO <= EXPRESSAO
    | EXPRESSAO >= EXPRESSAO
    | EXPRESSAO == EXPRESSAO
